on_birthday_childhood = {
    on_actions = {
        on_fae_birthday_childhood
    }
}

on_fae_birthday_childhood = {
    trigger = {
        OR = {
            is_adult = no
            age = 16
        }
        OR = {
            has_trait = fae_queen
            has_trait = fae_blood
        }
    }

    effect = {
        set_immortal_age = age
    }
}
